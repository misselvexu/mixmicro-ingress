== RateLimiter Config Properties

- 网关限流

[source,json]
----

[
  {
    "resource": "ws",
    "count": 1000
  },
  {
    "resource": "/demo/echo",
    "limitApp": "default",
    "grade": 1,
    "count": 5,
    "strategy": 0,
    "controlBehavior": 0,
    "clusterMode": false
  }
]

----

[source,json]
----

[

    "resource": "workshop",
    "limitApp": "default",
    "count": 500.0,
    "timeWindow": 10,
    "grade": 0,
    "rtSlowRequestAmount": 5,
    "minRequestAmount": 5
  }
]
